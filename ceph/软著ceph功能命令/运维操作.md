# 运维操作

```shell
#升级操作
daemon-upgrade host=ycc-controller1 target=mon
#其他运维操作
daemon-create host=ycc-controller1 target=osd.38
daemon-stop host=ycc-controller1 target=osd.38
daemon-start host=ycc-controller1 target=osd.38
daemon-restart host=ycc-controller1 target=osd.38
daemon-list host=ycc-controller1 target=osd.38
daemon-d host=ycc-controller1 target=osd.38
```
